    points = [(1, 1), (3, 2), (5, 7)] # A, B, C
    X = (2, 3)
    
    from math import sqrt
    
    def distMeasure(points, X):
        result = []
        for point in points:
            dist = sqrt((point[0] - X[0])**2 + (point[1] - X[1])**2)
            result.append(dist)
        return result
    distMeasure(points, X)
    # [2.23606797749979, 1.4142135623730951, 5.0]
